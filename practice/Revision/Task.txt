import java.util.*;

class Book {
    
    int bookId;
    String title;
    String author;
    
    Book(int bookId,String title,String author) {
        this.bookId = bookId;
        this.title = title;
        this.author = author;
    }
    
    public int getBookId() {
        return bookId;
    }
    
    
    public String toString() {
        return "ID: "+bookId+" | Title: "+title+" | Author: "+author;

    }
}

class BookCatalog {
    
    HashSet<Book> books = new HashSet<>();
    
    void removeBook(int bookId) {
        //if book is exisitng with this id then remove book and display "Book removed successfully."
        
        //if book is not existing display message "Book not found with id: "+bookId.
    }
    
    void addBook(Book book) {
        boolean isExisting = false;
        for(Book eb: books) {
            if(eb.getBookId() == book.getBookId()) {
                System.out.println("Duplicate book found with ID: "+book.getBookId());
                isExisting = true;
                break;
            }
        }
        if(isExisting == false) {
               books.add(book);
            System.out.println("Book added successfully."); 
        }
    }
    
    void displayAll() {
        for(Book b : books) {
            System.out.println(b);
        }
    }
}

class Main {
    
    public static void main(String[] args) {
        
        Scanner sc = new Scanner(System.in);
        
        int n = Integer.parseInt(sc.nextLine());
        
        BookCatalog catalog = new BookCatalog();
        
        for(int i=0;i < n;i++) {
            
            int id = Integer.parseInt(sc.nextLine());
            String title = sc.nextLine();
            String author = sc.nextLine();
            
            Book newBook = new Book(id, title,author);
            
            catalog.addBook(newBook);
        }
        
        catalog.displayAll();
    }
}
