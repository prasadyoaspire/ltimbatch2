import java.util.Scanner;

class Student {
    private int id;
    private String name;
    private String email;
    
    Student(int id,String name,String email) {
        this.id = id;
        this.name = name;
        this.email = email;
    }
    public void displayInfo() {
        System.out.println("Id= "+id+" name= "+name+" email= "+email);
    }
}

class StudentManager {
    
    private Student[] students;
    private int count;
    
    StudentManager(int capacity) {
        this.students = new Student[capacity];
        this.count = 0;
    }
    
    public void addStudent(int id, String name,String email) {
       students[this.count] = new Student(id,name,email); 
       this.count++;
    }
    
    public void displayAll() {
        for(int i=0;i<students.length;i++) {
            students[i].displayInfo();
        }
    }
}

class Main{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = Integer.parseInt(sc.nextLine()); //size of array
        
        StudentManager manager = new StudentManager(n);
        
        for(int i=0;i<n;i++) {
            int id = Integer.parseInt(sc.nextLine());
            String name = sc.nextLine();
            String email = sc.nextLine();
            manager.addStudent(id,name,email);
        }
        manager.displayAll();
    }
}
